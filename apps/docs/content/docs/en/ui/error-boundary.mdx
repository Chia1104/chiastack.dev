## Basic Usage

```tsx
import { ErrorBoundary } from '@chiastack/ui/error-boundary';

function App() {
  return (
    <ErrorBoundary
      fallback={({ error, reset }) => (
        <div>
          <h2>Something went wrong</h2>
          <p>{error?.message}</p>
          <button onClick={reset}>Retry</button>
        </div>
      )}
    >
      <YourComponent />
    </ErrorBoundary>
  );
}
```

## Props

- `children`: React.ReactNode - Child components
- `className?: string` - Custom CSS class name
- `fallback?: (({ error, reset }) => React.ReactNode) | React.ReactNode` - UI to display on error
- `onError?: (error: TError, errorInfo: React.ErrorInfo) => void` - Error catch callback
- `handleRetry?: ({ error, reset }) => void` - Retry handler function

## Advanced Usage

### Custom Error Handling

```tsx
import { ErrorBoundary } from '@chiastack/ui/error-boundary';

function App() {
  return (
    <ErrorBoundary
      fallback={({ error, reset }) => (
        <div className="error-container">
          <h2>Something went wrong</h2>
          <p>{error?.message}</p>
          <button onClick={reset}>Retry</button>
        </div>
      )}
      onError={(error, errorInfo) => {
        // Log error to error tracking service
        console.error('Error caught:', error, errorInfo);
        // Can send to Sentry, LogRocket, etc.
      }}
      handleRetry={({ error, reset }) => {
        console.log('Retrying after error:', error);
        // Perform cleanup before retry
        reset();
      }}
    >
      <YourComponent />
    </ErrorBoundary>
  );
}
```

### Using Functional Fallback

```tsx
import { ErrorBoundary } from '@chiastack/ui/error-boundary';

function App() {
  return (
    <ErrorBoundary
      fallback={({ error, reset }) => {
        if (error?.message.includes('Network')) {
          return (
            <div>
              <h2>Network Error</h2>
              <p>Please check your network connection</p>
              <button onClick={reset}>Retry</button>
            </div>
          );
        }
        return (
          <div>
            <h2>Unknown Error</h2>
            <p>{error?.message}</p>
            <button onClick={reset}>Retry</button>
          </div>
        );
      }}
    >
      <YourComponent />
    </ErrorBoundary>
  );
}
```

### Using Static Fallback

```tsx
import { ErrorBoundary } from '@chiastack/ui/error-boundary';

function App() {
  return (
    <ErrorBoundary fallback={<div>An error occurred, please refresh the page</div>}>
      <YourComponent />
    </ErrorBoundary>
  );
}
```

## Complete Example

```tsx
import { ErrorBoundary } from '@chiastack/ui/error-boundary';
import { Chart } from '@chiastack/ui/trading-chart/chart';

function TradingApp() {
  return (
    <ErrorBoundary
      fallback={({ error, reset }) => (
        <div className="error-boundary">
          <h2>Chart failed to load</h2>
          <p>Error: {error?.message}</p>
          <button onClick={reset}>Reload Chart</button>
        </div>
      )}
      onError={(error, errorInfo) => {
        // Log error
        console.error('Chart error:', error, errorInfo);
      }}
    >
      <Chart>
        {/* Chart components */}
      </Chart>
    </ErrorBoundary>
  );
}
```
